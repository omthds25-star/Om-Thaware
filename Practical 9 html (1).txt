<script>
    function loadPage(page) {
        let frame = document.getElementById('frame');

        if(page === 'home') {
            frame.srcdoc = `
                <h2 style='text-align:center;'>Welcome</h2>
                <p style='text-align:center;'>This is Home Page</p>
            `;
        }

        else if(page === 'map') {
            frame.src = 'https://www.google.com/maps?q=Nasik&output=embed';
        }

        else {
            frame.src = page;
        }
    }
</script>
